    getting_started

    A simple example of using the RF24 class

    examples/getting_started.py
      1"""
  2Simple example of using the RF24 class.
  3"""
  4import sys
  5import argparse
  6import time
  7import struct
  8from pyrf24 import RF24, RF24_PA_LOW
  9
 10
 11########### USER CONFIGURATION ###########
 12# See https://github.com/TMRh20/RF24/blob/master/pyRF24/readme.md
 13# Radio CE Pin, CSN Pin, SPI Speed
 14# CE Pin uses GPIO number with BCM and SPIDEV drivers, other platforms use
 15# their own pin numbering
 16# CS Pin addresses the SPI bus number at /dev/spidev<a>.<b>
 17# ie: RF24 radio(<ce_pin>, <a>*10+<b>); spidev1.0 is 10, spidev1.1 is 11 etc..
 18radio = RF24(22, 0)
 19
 20# using the python keyword global is bad practice. Instead we'll use a 1 item
 21# list to store our float number for the payloads sent
 22payload = [0.0]
 23
 24# For this example, we will use different addresses
 25# An address need to be a buffer protocol object (bytearray)
 26address = [b"1Node", b"2Node"]
 27# It is very helpful to think of an address as a path instead of as
 28# an identifying device destination
 29
 30# to use different addresses on a pair of radios, we need a variable to
 31# uniquely identify which address this radio will use to transmit
 32# 0 uses address[0] to transmit, 1 uses address[1] to transmit
 33radio_number = bool(
 34    int(input("Which radio is this? Enter '0' or '1'. Defaults to '0' ") or 0)
 35)
 36
 37# initialize the nRF24L01 on the spi bus
 38if not radio.begin():
 39    raise OSError("nRF24L01 hardware isn't responding")
 40
 41# set the Power Amplifier level to -12 dBm since this test example is
 42# usually run with nRF24L01 transceivers in close proximity of each other
 43radio.set_pa_level(RF24_PA_LOW)  # RF24_PA_MAX is default
 44
 45# set TX address of RX node into the TX pipe
 46radio.open_tx_pipe(address[radio_number])  # always uses pipe 0
 47
 48# set RX address of TX node into an RX pipe
 49radio.open_rx_pipe(1, address[not radio_number])  # using pipe 1
 50
 51# To save time during transmission, we'll set the payload size to be only what
 52# we need. A float value occupies 4 bytes in memory using struct.calcsize()
 53# "<f" means a little endian unsigned float
 54radio.payload_size = struct.calcsize("<f")
 55
 56# for debugging
 57radio.print_details()
 58
 59
 60def master(count: int = 5):  # count = 5 will only transmit 5 packets
 61    """Transmits an incrementing float every second"""
 62    radio.listen = False  # ensures the nRF24L01 is in TX mode
 63
 64    while count:
 65        # use struct.pack() to pack your data into a usable payload
 66        # into a usable payload
 67        buffer = struct.pack("<f", payload[0])
 68        # "<f" means a single little endian (4 byte) float value.
 69        start_timer = time.monotonic_ns()  # start timer
 70        result = radio.write(buffer)
 71        end_timer = time.monotonic_ns()  # end timer
 72        if not result:
 73            print("Transmission failed or timed out")
 74        else:
 75            print(
 76                "Transmission successful! Time to Transmit:",
 77                f"{(end_timer - start_timer) / 1000} us. Sent: {payload[0]}",
 78            )
 79            payload[0] += 0.01
 80        time.sleep(1)
 81        count -= 1
 82
 83
 84def slave(timeout: int = 6):
 85    """Polls the radio and prints the received value. This method expires
 86    after 6 seconds of no received transmission."""
 87    radio.listen = True  # put radio into RX mode and power up
 88
 89    start = time.monotonic()
 90    while (time.monotonic() - start) < timeout:
 91        has_payload, pipe_number = radio.available_pipe()
 92        if has_payload:
 93            length = radio.payload_size  # grab the payload length
 94            # fetch 1 payload from RX FIFO
 95            received = radio.read(length)  # also clears radio.irq_dr status flag
 96            # expecting a little endian float, thus the format string "<f"
 97            # received[:4] truncates padded 0s in case dynamic payloads are disabled
 98            payload[0] = struct.unpack("<f", received[:4])[0]
 99            # print details about the received packet
100            print(f"Received {length} bytes on pipe {pipe_number}: {payload[0]}")
101            start = time.monotonic()  # reset the timeout timer
102
103    # recommended behavior is to keep in TX mode while idle
104    radio.listen = False  # put the nRF24L01 is in TX mode

    acknowledgement_payloads

    A simple example of using the RF24 class to attach an Acknowledgement (ACK) payload the automatically generated ACK packet.

    examples/acknowledgement_payloads.py
      1"""
  2Simple example of using the library to transmit
  3and retrieve custom automatic acknowledgment payloads.
  4"""
  5import sys
  6import time
  7import argparse
  8from pyrf24 import RF24, RF24_PA_LOW

 20# list to store our integer number for the payloads' counter
 21counter = [0]
 22
 23# For this example, we will use different addresses
 24# An address need to be a buffer protocol object (bytearray)
 25address = [b"1Node", b"2Node"]

 42radio.set_pa_level(RF24_PA_LOW)  # RF24_PA_MAX is default
 43
 44# ACK payloads are dynamically sized, so we need to enable that feature also
 45radio.dynamic_payloads = True
 46
 47# to enable the custom ACK payload feature
 48radio.ack_payloads = True
 49
 50# set TX address of RX node into the TX pipe
 51radio.open_tx_pipe(address[radio_number])  # always uses pipe 0
 52
 53# set RX address of TX node into an RX pipe

 55

 57radio.print_pretty_details()

 60def master(count: int = 5):  # count = 5 will only transmit 5 packets
 61    """Transmits a payload every second and prints the ACK payload"""
 62    radio.listen = False  # put radio in TX mode
 63
 64    while count:
 65        # construct a payload to send
 66        buffer = b"Hello \x00" + bytes([counter[0]])
 67
 68        # send the payload and prompt
 69        start_timer = time.monotonic_ns()  # start timer
 70        result = radio.write(buffer)  # save the report
 71        end_timer = time.monotonic_ns()  # stop timer
 72        if result:
 73            # print timer results upon transmission success
 74            print(
 75                "Transmission successful! Time to transmit:",
 76                f"{int((end_timer - start_timer) / 1000)} us. Sent:",
 77                f"{buffer[:6].decode('utf-8')}{counter[0]}",
 78                end=" ",
 79            )
 80            if radio.available():
 81                # print the received ACK that was automatically sent
 82                result = radio.read(radio.get_dynamic_payload_size())
 83                print(f" Received: {result[:6].decode('utf-8')}{result[7:8][0]}")
 84                counter[0] += 1  # increment payload counter
 85            else:
 86                print(" Received an empty ACK packet")
 87        else:
 88            print("Transmission failed or timed out")
 89        time.sleep(1)  # let the RX node prepare a new ACK payload
 90        count -= 1
 91
 92
 93def slave(timeout: int = 6):
 94    """Prints the received value and sends an ACK payload"""
 95    radio.listen = True  # put radio into RX mode, power it up
 96
 97    # setup the first transmission's ACK payload
 98    buffer = b"World \x00" + bytes([counter[0]])
 99    # we must set the ACK payload data and corresponding
100    # pipe number [0,5]
101    radio.write_ack_payload(1, buffer)  # load ACK for first response
102
103    start = time.monotonic()  # start timer
104    while (time.monotonic() - start) < timeout:
105        has_payload, pipe_number = radio.available_pipe()
106        if has_payload:
107            length = radio.get_dynamic_payload_size()  # grab the payload length
108            received = radio.read(length)  # fetch 1 payload from RX FIFO
109            # increment counter from received payload
110            counter[0] = received[7:8][0] + 1
111            print(
112                f"Received {length} bytes on pipe {pipe_number}:",
113                f"{received[:6].decode('utf-8')}{received[7:8][0]} Sent:",
114                f"{buffer[:6].decode('utf-8')}{counter[0]}",
115            )
116            start = time.monotonic()  # reset timer
117
118            # build a new ACK payload
119            buffer = b"World \x00" + bytes([counter[0]])
120            radio.write_ack_payload(1, buffer)  # load ACK for next response
121
122    # recommended behavior is to keep in TX mode while idle
123    radio.listen = False  # put radio in TX mode & flush unused ACK payloads

    manual_acknowledgements

    A simple example of using the RF24 class to receive a payload and then send a responding payload as an Acknowledgement. This functionality is similar to using the nRF24L01s ACK payload feature, but doe not append ACK payloads to the ACK packets.

    examples/manual_acknowledgements.py
      1"""
  2Simple example of using the RF24 class to transmit and respond with
  3acknowledgment (ACK) transmissions. Notice that the auto-ack feature is
  4enabled, but this example doesn't use automatic ACK payloads because automatic
  5ACK payloads' data will always be outdated by 1 transmission. Instead, this
  6example uses a call and response paradigm.
  7"""
  8import sys
  9import argparse
 10import time
 11import struct
 12from pyrf24 import RF24, RF24_PA_LOW
 13
 14########### USER CONFIGURATION ###########
 15# See https://github.com/TMRh20/RF24/blob/master/pyRF24/readme.md
 16# Radio CE Pin, CSN Pin, SPI Speed
 17# CE Pin uses GPIO number with BCM and SPIDEV drivers, other platforms use
 18# their own pin numbering
 19# CS Pin addresses the SPI bus number at /dev/spidev<a>.<b>
 20# ie: RF24 radio(<ce_pin>, <a>*10+<b>); spidev1.0 is 10, spidev1.1 is 11 etc..
 21radio = RF24(22, 0)
 22
 23# For this example, we will use different addresses
 24# An address need to be a buffer protocol object (bytearray)
 25address = [b"1Node", b"2Node"]
 26# It is very helpful to think of an address as a path instead of as
 27# an identifying device destination
 28
 29# to use different addresses on a pair of radios, we need a variable to
 30# uniquely identify which address this radio will use to transmit
 31# 0 uses address[radio_number] to transmit, 1 uses address[not radio_number] to transmit
 32radio_number = bool(
 33    int(input("Which radio is this? Enter '0' or '1'. Defaults to '0' ") or 0)
 34)
 35
 36# initialize the nRF24L01 on the spi bus
 37if not radio.begin():
 38    raise OSError("nRF24L01 hardware isn't responding")
 39
 40# set the Power Amplifier level to -12 dBm since this test example is
 41# usually run with nRF24L01 transceivers in close proximity of each other
 42radio.set_pa_level(RF24_PA_LOW)  # RF24_PA_MAX is default
 43
 44# set TX address of RX node into the TX pipe
 45radio.open_tx_pipe(address[radio_number])  # always uses pipe 0
 46
 47# set RX address of TX node into an RX pipe
 48radio.open_rx_pipe(1, address[not radio_number])  # using pipe 1
 49
 50# To save time during transmission, we'll set the payload size to be only what
 51# we need. A float value occupies 4 bytes in memory using struct.calcsize()
 52# "<b" means a little endian unsigned byte
 53# we also need an addition 7 bytes for the payload message
 54radio.payload_size = struct.calcsize("<b") + 7
 55
 56# for debugging
 57radio.print_pretty_details()
 58
 59# using the python keyword global is bad practice. Instead we'll use a 1 item
 60# list to store our float number for the payloads sent
 61payload = [0]
 62
 63
 64def master(count: int = 10):
 65    """Transmits a message and an incrementing integer every second"""
 66    radio.listen = False  # ensures the nRF24L01 is in TX mode
 67
 68    while count:  # only transmit `count` packets
 69        # use struct.pack() to pack your data into a usable payload
 70        # "<b" means a single little endian unsigned byte.
 71        # NOTE we added a b"\x00" byte as a c-string's NULL terminating 0
 72        buffer = b"Hello \x00" + struct.pack("<b", payload[0])
 73        start_timer = time.monotonic_ns()  # start timer
 74        result = radio.write(buffer)
 75        if not result:
 76            print("Transmission failed or timed out")
 77        else:
 78            radio.listen = True
 79            timout = time.monotonic() * 1000 + 200  # use 200 ms timeout
 80            ack = b"\x00" * len(buffer)  # variable used for the response
 81            while ack[0] == 0 and time.monotonic() * 1000 < timout:
 82                if radio.available():
 83                    # get the response & save it to ack variable
 84                    ack = radio.read(radio.payload_size)
 85            end_timer = time.monotonic_ns()  # end timer
 86            radio.listen = False
 87            print(
 88                "Transmission successful. Sent: ",
 89                f"{buffer[:6].decode('utf-8')}{payload[0]}.",
 90                end=" ",
 91            )
 92            if ack[0] == 0:
 93                print("No response received.")
 94            else:
 95                # decode response's text as an string
 96                # NOTE ack[:6] ignores the NULL terminating 0
 97                response = ack[:6].decode("utf-8")
 98                # use struct.unpack() to get the repsonse's appended int
 99                # NOTE ack[7:] discards NULL terminating 0, and
100                # "<b" means its a single little endian unsigned byte
101                payload[0] = struct.unpack("<b", ack[7:])[0]
102                print(
103                    f"Received: {response}{payload[0]}. Roundtrip delay:",
104                    f"{(end_timer - start_timer) / 1000} us.",
105                )
106        time.sleep(1)  # make example readable by slowing down transmissions
107        count -= 1
108
109
110def slave(timeout: int = 6):
111    """Polls the radio and prints the received value. This method expires
112    after 6 seconds of no received transmission"""
113    radio.listen = True  # put radio into RX mode and power up
114
115    start_timer = time.monotonic()  # start a timer to detect timeout
116    while (time.monotonic() - start_timer) < timeout:
117        # receive payloads or wait 6 seconds till timing out
118        has_payload, pipe_number = radio.available_pipe()
119        if has_payload:
120            length = radio.payload_size  # grab the payload length
121            received = radio.read(length)  # fetch 1 payload from RX FIFO
122            # use struct.unpack() to get the payload's appended int
123            # NOTE received[7:] discards NULL terminating 0, and
124            # "<b" means its a single little endian unsigned byte
125            payload[0] = struct.unpack("<b", received[7:])[0] + 1
126            # use bytes() to pack our data into a usable payload
127            # NOTE b"\x00" byte is a c-string's NULL terminating 0
128            buffer = b"World \x00" + bytes([payload[0]])
129            radio.listen = False  # set radio to TX mode
130            radio.write_fast(buffer)  # load payload into radio's RX buffer
131            # keep retrying to send response for 150 milliseconds
132            response = radio.tx_standby(150)  # save response's result
133            radio.listen = True  # set radio back into RX mode
134            # print the payload received and the response's payload
135            print(
136                f"Received {length} bytes on pipe {pipe_number}:",
137                f"{received[:6].decode('utf-8')}{payload[0] - 1}.",
138                end=" ",
139            )
140            if response:
141                print(f"Sent: {buffer[:6].decode('utf-8')}{payload[0]}")
142            else:
143                print("Response failed or timed out")
144            start_timer = time.monotonic()  # reset the timeout timer
145
146    # recommended behavior is to keep in TX mode while idle
147    radio.listen = False  # put the nRF24L01 is in TX mode

    multiceiver_demo

    A simple example of using the RF24 class to receive data from multiple transmitting nRF24L01 transceivers. This technique is trademarked by Nordic Semiconductors as MulticeiverTM

    examples/multiceiver_demo.py
      1"""
  2Simple example of using 1 nRF24L01 to receive data from up to 6 other
  3transceivers. This technique is called "multiceiver" in the datasheet.
  4"""
  5import sys
  6import argparse
  7import time
  8import struct
  9from pyrf24 import RF24, RF24_PA_LOW
 10
 11########### USER CONFIGURATION ###########
 12# See https://github.com/TMRh20/RF24/blob/master/pyRF24/readme.md
 13# Radio CE Pin, CSN Pin, SPI Speed
 14# CE Pin uses GPIO number with BCM and SPIDEV drivers, other platforms use
 15# their own pin numbering
 16# CS Pin addresses the SPI bus number at /dev/spidev<a>.<b>
 17# ie: RF24 radio(<ce_pin>, <a>*10+<b>); spidev1.0 is 10, spidev1.1 is 11 etc..
 18radio = RF24(22, 0)
 19
 20# setup the addresses for all transmitting nRF24L01 nodes
 21addresses = [
 22    b"\x78" * 5,
 23    b"\xF1\xB6\xB5\xB4\xB3",
 24    b"\xCD\xB6\xB5\xB4\xB3",
 25    b"\xA3\xB6\xB5\xB4\xB3",
 26    b"\x0F\xB6\xB5\xB4\xB3",
 27    b"\x05\xB6\xB5\xB4\xB3",
 28]

 50def master(node_number: int = 0, count: int = 6):
 51    """start transmitting to the base station.
 52
 53    :param int node_number: the node's identifying index (from the
 54        the `addresses` list). This is a required parameter
 55    :param int count: the number of times that the node will transmit
 56        to the base station.
 57    """
 58    # According to the datasheet, the auto-retry features's delay value should
 59    # be "skewed" to allow the RX node to receive 1 transmission at a time.
 60    # So, use varying delay between retry attempts and 15 (at most) retry attempts
 61    radio.set_retries(((node_number * 3) % 12) + 3, 15)  # max value is 15 for both args
 62
 63    radio.listen = False
 64    # set the TX address to the address of the base station.
 65    radio.open_tx_pipe(addresses[node_number])
 66    counter = 0
 67    # use the node_number to identify where the payload came from
 68    while counter < count:
 69        counter += 1
 70        # payloads will include the node_number and a payload ID character
 71        payload = struct.pack("<ii", node_number, counter)
 72        start_timer = time.monotonic_ns()
 73        report = radio.write(payload)
 74        end_timer = time.monotonic_ns()
 75        # show something to see it isn't frozen
 76        if report:
 77            print(
 78                f"Transmission of payloadID {counter} as node {node_number}",
 79                f"successfull! Transmission time: {(end_timer - start_timer) / 1000}",
 80                "us",
 81            )
 82        else:
 83            print("Transmission failed or timed out")
 84        time.sleep(0.5)  # slow down the test for readability

 87def slave(timeout=10):
 88    """Use the nRF24L01 as a base station for listening to all nodes"""
 89    # write the addresses to all pipes.
 90    for pipe_n, addr in enumerate(addresses):
 91        radio.open_rx_pipe(pipe_n, addr)
 92    radio.listen = True  # put base station into RX mode
 93    start_timer = time.monotonic()  # start timer

 95        has_payload, pipe_number = radio.available_pipe()
 96        if has_payload:
 97            length = radio.payload_size  # grab the payload length

 50def make_buffer(buf_iter: int, size: int = 32):
 51    """return a list of payloads"""
 52    # we'll use `size` for the number of payloads in the list and the
 53    # payloads' length
 54    # prefix payload with a sequential letter to indicate which
 55    # payloads were lost (if any)
 56    buff = bytes([buf_iter + (65 if 0 <= buf_iter < 26 else 71)])
 57    for j in range(size - 1):
 58        char = bool(j >= (size - 1) / 2 + abs((size - 1) / 2 - buf_iter))
 59        char |= bool(j < (size - 1) / 2 - abs((size - 1) / 2 - buf_iter))
 60        buff += bytes([char + 48])
 61    return buff

 71    #  number of bytes we need to transmit

 78        buf_iter = 0  # iterator of payloads for the while loop
 79        failures = 0  # keep track of manual retries

 82            buf = make_buffer(buf_iter, size)  # make a payload

 85                failures += 1  # increment manual retries

 88                    # we need to prevent an infinite loop
 89                    print(
 90                        "Make sure slave() node is listening." " Quiting master_fifo()"
 91                    )
 92                    buf_iter = size + 1  # be sure to exit the while loop
 93                    radio.flush_tx()  # discard all payloads in TX FIFO
 94                    break
 95            buf_iter += 1
 96        end_timer = time.monotonic() * 1000  # end timer
 97        print(
 98            f"Transmission took {end_timer - start_timer} ms with",
 99            f"{failures} failures detected.",

 75# This example covers fragmented payloads also. Set a sentinel for readability.
 76MAX_FRAG_SIZE = 24

 79# list to store our number of the payloads sent

 83def idle(timeout: int = 30, strict_timeout: bool = False):
 84    """Listen for any payloads and print the transaction
 85
 86    :param int timeout: The number of seconds to wait (with no transmission)
 87        until exiting function.
 88    :param bool strict_timeout: If set to True, then the timer is not reset when
 89        processing incoming traffic
 90    """
 91    print("idling for", timeout, "seconds")
 92    start_timer = time.monotonic()
 93    while (time.monotonic() - start_timer) < timeout:
 94        if IS_MESH and mesh.update() and not mesh.node_id:
 95            # if this is a mesh master node and update() returned a non-zero value
 96            mesh.dhcp()  # only needed for master node
 97        else:
 98            network.update()  # keep the network layer current
 99        while network.available():
100            if not strict_timeout:
101                start_timer = time.monotonic()  # reset timer
102            header, payload = network.read()
103            payload_len = len(payload)
104            print("Received payload", end=" ")
105            # C++ examples only use 1 or 2 long ints as small messages
106            if payload_len < MAX_FRAG_SIZE and payload_len % 4 == 0:
107                # if not a large fragmented message and multiple of 4 bytes
108                fmt = "<" + "L" * int(payload_len / 4)
109                print(struct.unpack(fmt, payload), end=" ")
110            print(header.to_string(), "length", payload_len)
111

113def emit(
114    node: int = not THIS_NODE, frag: bool = False, count: int = 5, interval: int = 1
115):
116    """Transmits 1 (or 2) integers or a large buffer
117
118    :param int node: The target node for network transmissions.
119        If using RF24Mesh, this is a unique node_id.
120        If using RF24Network, this is the node's logical address.
121    :param bool frag: Only use fragmented messages?
122    :param int count: The max number of messages to transmit.
123    :param int interval: time (in seconds) between transmitting messages.
124    """
125    while count:
126        # for this example, we use idle() to update() the network/mesh layer
127        idle(interval, True)  # idle till its time to emit
128        count -= 1
129        packets_sent[0] += 1
130        # TMRh20's RF24Mesh examples use 1 long int containing a timestamp (in ms)
131        message = struct.pack("<L", int(time.monotonic() * 1000))
132        if frag:
133            message = bytes(range((packets_sent[0] + MAX_FRAG_SIZE) % MAX_PAYLOAD_SIZE))
134        elif not IS_MESH:  # if using RF24Network
135            # TMRh20's RF24Network examples use 2 long ints, so add another
136            message += struct.pack("<L", packets_sent[0])
137        result = False
138        start = time.monotonic_ns()
139        if IS_MESH:  # write() is a little different for RF24Mesh vs RF24Network
140            result = mesh.write(message, ord("M"), node)
141        else:
142            result = network.write(RF24NetworkHeader(node, ord("T")), message)
143        end = time.monotonic_ns()
144        print(
145            f"Sending {packets_sent[0]} (len {len(message)})...",
146            "ok." if result else "failed.",
147            f"Transmission took {int((end - start) / 1000000)} ms",
148        )
